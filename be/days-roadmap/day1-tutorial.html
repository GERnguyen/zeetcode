  provider  = "postgresql"
  url       = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")

enum PrivacySetting {
  PUBLIC
  FOLLOWERS_ONLY
  PRIVATE
}

enum FollowStatus {
  PENDING
  ACCEPTED
  REJECTED
}

enum UserRole {
  USER
  ADMIN
}

model User {
  id                String         @id @default(uuid())
  username          String         @unique
  email             String         @unique
  password          String
  role              UserRole       @default(USER)
  profileVisibility PrivacySetting @default(PUBLIC)
  createdAt         DateTime       @default(now())

  following         Followers[]    @relation("UserAsFollower")
  followers         Followers[]    @relation("UserAsFollowing")
}

model Followers {
  followerId  String
  followingId String
  status      FollowStatus @default(PENDING)
  createdAt   DateTime     @default(now())

  follower    User         @relation("UserAsFollower", fields: [followerId], references: [id], onDelete: Cascade)
  following   User         @relation("UserAsFollowing", fields: [followingId], references: [id], onDelete: Cascade)

  @@id([followerId, followingId])
}

Thực thi câu lệnh để đẩy cấu trúc lên Supabase và sinh code Prisma Client:

npx prisma migrate dev --name init_user_rbac_follower
npx prisma generate

Tầng Middlewares (Bảo mật & Phân quyền)

src/middlewares/auth.middleware.ts
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

export const requireAuth = (req: Request, res: Response, next: NextFunction) => {
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ success: false, message: "Thiếu Token xác thực!" });

    try {
        const decoded = jwt.verify(token, process.env.JWT_SECRET || 'secret');
        (req as any).user = decoded;
        next();
    } catch (error) {
        return res.status(401).json({ success: false, message: "Token không hợp lệ!" });
    }
};
src/middlewares/admin.middleware.ts
import { Request, Response, NextFunction } from 'express';

export const requireAdmin = (req: Request, res: Response, next: NextFunction) => {
    const user = (req as any).user;
    if (user?.role !== 'ADMIN') {
        return res.status(403).json({ 
            success: false, 
            message: "Chỉ Admin mới có quyền thực hiện hành động này!" 
        });
    }
    next();
};

Tầng Repository

src/repositories/user.repository.ts
import { prisma } from '../config/prisma.config';
import { FollowStatus } from '../generated/prisma';

export class UserRepository {
    async createUser(data: any) {
        return await prisma.user.create({ data });
    }

    async findByEmail(email: string) {
        return await prisma.user.findUnique({ where: { email } });
    }

    async findById(id: string) {
        return await prisma.user.findUnique({ where: { id } });
    }

    async requestFollow(followerId: string, followingId: string) {
        return await prisma.followers.upsert({
            where: { followerId_followingId: { followerId, followingId } },
            update: { status: FollowStatus.PENDING },
            create: { followerId, followingId }
        });
    }

    async updateFollowStatus(followerId: string, followingId: string, status: FollowStatus) {
        return await prisma.followers.update({
            where: { followerId_followingId: { followerId, followingId } },
            data: { status }
        });
    }

    async checkActiveRelation(followerId: string, followingId: string) {
        const relation = await prisma.followers.findUnique({
            where: { followerId_followingId: { followerId, followingId } }
        });
        return relation?.status === FollowStatus.ACCEPTED;
    }
}

Tầng Service

src/services/user.service.ts
import { UserRepository } from '../repositories/user.repository';
import { FollowStatus } from '../generated/prisma';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';

export class UserService {
    private userRepo = new UserRepository();

    async register(data: any) {
        const passwordHash = await bcrypt.hash(data.password, 10);
        return await this.userRepo.createUser({
            username: data.username,
            email: data.email,
            password: passwordHash
        });
    }

    async login(data: any) {
        const user = await this.userRepo.findByEmail(data.email);
        if (!user || !(await bcrypt.compare(data.password, user.password))) {
            throw new Error("Thông tin đăng nhập không hợp lệ!");
        }

        const token = jwt.sign(
            { id: user.id, role: user.role }, 
            process.env.JWT_SECRET || 'secret', 
            { expiresIn: '1d' }
        );
        return { token, user: { id: user.id, username: user.username, role: user.role } };
    }

    async requestFollow(followerId: string, followingId: string) {
        if (followerId === followingId) throw new Error("Không thể tự follow chính mình!");
        return await this.userRepo.requestFollow(followerId, followingId);
    }

    async respondFollow(followerId: string, followingId: string, action: 'ACCEPT' | 'REJECT') {
        const status = action === 'ACCEPT' ? FollowStatus.ACCEPTED : FollowStatus.REJECTED;
        return await this.userRepo.updateFollowStatus(followerId, followingId, status);
    }
}

4 Khởi chạy Server

Cập nhật file src/server.ts (tương tự như các service khác) và liên kết các router lại với nhau. Sau đó thêm script vào package.json:

"scripts": {
  "dev": "nodemon src/server.ts"
}

Cuối cùng, khởi động dịch vụ:

npm run dev
